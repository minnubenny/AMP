class FocalLoss(nn.Module):
    def __init__(self, alpha=0.25, gamma=2):
        super().__init__()
        self.alpha = alpha
        self.gamma = gamma
        self.bce = nn.BCEWithLogitsLoss(reduction="none")

    def forward(self, inputs, targets):
        bce_loss = self.bce(inputs, targets)
        probs = torch.sigmoid(inputs)
        pt = targets * probs + (1 - targets) * (1 - probs)
        focal = self.alpha * (1 - pt) ** self.gamma * bce_loss
        return focal.mean()

VALID_AA = set("ACDEFGHIKLMNPQRSTVWYX")

def clean_sequence(seq):
    seq = str(seq).strip().upper()
    if seq.startswith(">"):
        return None
    seq = "".join([aa if aa in VALID_AA else "X" for aa in seq])
    if len(seq) < 5:
        return None
    return seq
    
    
    
    
    
"#Dataset#"
class MFTPDataset(Dataset):

    def __init__(self, csv_path, label_cols):

        df = pd.read_csv(csv_path)

        seqs = []
        labels = []

        for _, row in df.iterrows():

            seq = clean_sequence(row["sequence"])

            if seq is not None:
                seqs.append(seq)
                labels.append(row[label_cols].values.astype("float32"))

        self.seqs = seqs
        self.labels = torch.from_numpy(np.array(labels)).float()

    def __len__(self):
        return len(self.seqs)

    def __getitem__(self, idx):
        return self.seqs[idx], self.labels[idx]

"#CBAM#"
class CBAM(nn.Module):
    def __init__(self, channels):
        super().__init__()
        self.mlp = nn.Sequential(
            nn.Linear(channels, channels // 8),
            nn.ReLU(),
            nn.Linear(channels // 8, channels)
        )
        self.conv = nn.Conv1d(2, 1, 7, padding=3)

    def forward(self, x):
        avg_pool = torch.mean(x, dim=2)
        max_pool, _ = torch.max(x, dim=2)
        x = x * torch.sigmoid(self.mlp(avg_pool) + self.mlp(max_pool)).unsqueeze(2)

        avg = torch.mean(x, dim=1, keepdim=True)
        max_, _ = torch.max(x, dim=1, keepdim=True)
        return x * torch.sigmoid(self.conv(torch.cat([avg, max_], dim=1)))
        
        
        
        
        
        
"#Model (Supports Ablation)#"
class AMPModel(nn.Module):

    def __init__(self, num_labels):
        super().__init__()

        self.cnn = nn.Conv1d(320, 320, 5, padding=2)
        self.bilstm = nn.LSTM(320, 128, batch_first=True, bidirectional=True)

        if USE_CBAM:
            self.cbam = CBAM(256)

        self.dropouts = nn.ModuleList([nn.Dropout(0.5) for _ in range(5)])
        self.attn = nn.Sequential(
            nn.Linear(256,128),
            nn.Tanh(),
            nn.Linear(128,1)
        )

        self.label_emb = nn.Embedding(num_labels, 256)
        self.temperature = nn.Parameter(torch.ones(num_labels))
    def forward(self, tokens):

        x = esm_model(tokens, repr_layers=[6])["representations"][6]

        # CNN
        x = F.relu(self.cnn(x.permute(0,2,1)))

        # LSTM
        x, _ = self.bilstm(x.permute(0,2,1))

        # CBAM
        if USE_CBAM:
            x = self.cbam(x.permute(0,2,1))

        # Attention pooling
        attn_weights = torch.softmax(self.attn(x.permute(0,2,1)), dim=1)
        peptide_emb = torch.sum(attn_weights * x.permute(0,2,1), dim=1)

        logits = 0
        for dropout in self.dropouts:
            logits += dropout(peptide_emb) @ self.label_emb.weight.T

        logits = logits / len(self.dropouts)
        logits = logits * self.temperature

        return logits, peptide_emb

for fold, (train_idx, val_idx) in enumerate(kf.split(dataset)):
    print(f"\n===== Fold {fold+1} =====")

    train_ds = Subset(dataset, train_idx)
    val_ds = Subset(dataset, val_idx)

    train_loader = DataLoader(train_ds, batch_size=8, shuffle=True)
    val_loader = DataLoader(val_ds, batch_size=8)

    model = AMPModel(len(label_columns)).to(device)

    pos = dataset.labels.sum(dim=0)
    neg = len(dataset) - pos
    pos_weight = (neg / (pos + 1e-6)).to(device)

    criterion = FocalLoss(alpha=0.5, gamma=2)
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-4)
    
    # TRAIN
    for epoch in range(35):

        model.train()
        total_loss = 0   # track loss

        for seqs, labels in train_loader:
            data = [("", s) for s in seqs]
            _, _, tokens = batch_converter(data)
            tokens = tokens[:, :150].to(device)
            labels = labels.to(device)

            logits, pep_emb = model(tokens)
            loss = criterion(logits, labels)

            if USE_SEMANTIC:
                label_ids = torch.arange(len(label_columns)).to(device)
                label_emb = model.label_emb(label_ids)
                sim = pep_emb @ label_emb.T
                sem_loss = F.binary_cross_entropy_with_logits(sim, labels)
                loss += 0.15 * sem_loss

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            total_loss += loss.item()
            
             # ✅ PRINT AFTER EACH EPOCH
        avg_loss = total_loss / len(train_loader)

        print(f"Fold {fold+1} | Epoch {epoch+1}/35 | Loss: {avg_loss:.4f}")
        
   



model.eval()

def predict(sequence, model):
    sequence = clean_sequence(sequence)
    data = [("", sequence)]
    _, _, tokens = batch_converter(data)
    tokens = tokens[:, :150].to(device)

    model.eval()
    with torch.no_grad():
        logits, _ = model(tokens)
        probs = torch.sigmoid(logits)[0].cpu().numpy()

    if USE_THRESHOLD:
        return {
            label: int(probs[i] > thresholds[i])
            for i, label in enumerate(label_columns)
        }
    else:
        return {
            label: int(probs[i] > 0.5)
            for i, label in enumerate(label_columns)
        }
